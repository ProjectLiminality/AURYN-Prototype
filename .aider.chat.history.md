def read_input_file(file_path):
    with open(file_path, 'r') as file:
        return file.read().strip()

def generate_project_name_and_prompt(input_text):
    io = InputOutput()
    chat = aider.ChatSession(io)
    
    system_message = """You are Auryn, an AI assistant that helps create new projects. 
    Based on the user's input, you need to:
    1. Generate a suitable project name (one or two words, CamelCase).
    2. Create an initial prompt for Aider to start the project implementation.
    Respond with only these two items, separated by a newline."""

    user_message = f"Project idea: {input_text}"
    
    response = chat.send_message(system_message + "\n\n" + user_message)
    project_name, initial_prompt = response.split('\n', 1)
    return project_name.strip(), initial_prompt.strip()

def create_repository(project_name):
    template_url = "https://github.com/yourusername/DreamNodeTemplate.git"
    subprocess.run(["git", "clone", template_url, project_name])
    os.chdir(project_name)
    subprocess.run(["git", "remote", "remove", "origin"])

def start_aider_session(initial_prompt):
    io = InputOutput()
    chat = aider.ChatSession(io)
    chat.send_message(initial_prompt)

def main():
    if len(sys.argv) != 2:
        print("Usage: python auryn.py <path_to_input_file>")
        sys.exit(1)

    input_file = Path(sys.argv[1])
    if not input_file.is_file():
        print(f"Error: {input_file} is not a valid file.")
        sys.exit(1)

    input_text = read_input_file(input_file)
    project_name, initial_prompt = generate_project_name_and_prompt(input_text)
    
    create_repository(project_name)
    start_aider_session(initial_prompt)

if __name__ == "__main__":
